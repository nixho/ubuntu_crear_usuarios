# Use the official Alpine Linux image as the base image
FROM alpine:latest

# Update the package repository and install required packages
RUN apk update && \
    apk upgrade && \
    apk add --no-cache \
    bash \
    curl \
    php \
    php-fpm \
    php-cli \
    php-json \
    php-phar \
    php-zip \
    php-mysqli \
    php-pdo_mysql \
    php-openssl \
    php-curl \
    php-dom \
    php-xml \
    php-mbstring \
    php-tokenizer \
    php-xmlwriter \
    php-simplexml \
    php-xmlreader \
    php-pear \
    php-ctype \
    php-fileinfo \
    php-gd \    
    php-exif \
    php-iconv \
    php-session 

# Installing composer
RUN curl -sS https://getcomposer.org/installer -o composer-setup.php \
    && php composer-setup.php --install-dir=/usr/local/bin --filename=composer \
    && rm composer-setup.php

RUN mkdir -p /home/tmp && chmod -R 755 /home/tmp \
    && chown -R nobody:nobody /home/tmp && chown -R nobody:nobody /var/log/php82

USER root

EXPOSE 9000

WORKDIR /var/www/

# Start PHP-FPM
CMD [ "php-fpm82","-F" ]
